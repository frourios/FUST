\chapter{Neutrality with Respect to the Axiom of Choice}
\label{app:ac_neutrality}

FUST is \textbf{neutral} with respect to the full Axiom of Choice (AC). The core structures of FUST are constructible within ZF + DC (Dependent Choice).

\begin{theorem}[Constructibility within ZF + DC]
\label{thm:zf_dc_constructible}
The core structures of FUST ($D_\zeta$, component operators $D_n$, kernels, gauge groups) are constructible within ZF + DC.
\end{theorem}

\begin{proof}
\begin{enumerate}
\item \textbf{$\varphi, \psi, \zeta_6$}: Roots of $x^2 - x - 1 = 0$ and $x^2 - x + 1 = 0$ as algebraic numbers. Constructively definable.
\item \textbf{$D_\zeta$ and $D_n$ coefficients}: Finite tuples of algebraic numbers.
\item \textbf{Kernels $\ker(D_n)$}: Finite-dimensional vector spaces (dim 1, 2, 3). No AC needed.
\item \textbf{Gauge groups}: SU(2), SU(3), U(1) are concretely constructible finite-dimensional Lie groups.
\end{enumerate}
\end{proof}

\begin{corollary}[AC Neutrality]
FUST is neutral with respect to AC:
\begin{itemize}
\item Full AC cannot be derived from ZF + DC + FUST
\item ZF + DC + $\neg$AC + FUST is also consistent (if ZF + DC + $\neg$AC is consistent)
\end{itemize}
\end{corollary}
