\section{\texorpdfstring{$\varphi$}{Ï†}-Scaling and the Step Constant}
\label{sec:phi-scaling}

\subsection{The Fundamental Step}

The $\varphi$-scaling operation $x \mapsto \varphi x$ becomes a translation in Frourio coordinates.

\begin{definition}[$\varphi$-Step]
\label{def:phi-step}
\lean{FUST.FrourioLogarithm.phiStep}
\leanok
\[
\mathrm{phiStep} := \log_{\mathfrak{f}}(\varphi) = \frac{\log \varphi}{\log \mathfrak{f}}
\]
\end{definition}

\begin{theorem}[$\varphi$-Scaling as Translation]
\label{thm:phi-scale-translation}
\lean{FUST.FrourioLogarithm.phi_scale_is_translation}
\leanok
For $x > 0$:
\[
\log_{\mathfrak{f}}(\varphi \cdot x) = \log_{\mathfrak{f}}(x) + \mathrm{phiStep}
\]
\end{theorem}

\begin{proof}
By logarithm properties:
\[
\log_{\mathfrak{f}}(\varphi \cdot x) = \frac{\log(\varphi \cdot x)}{\log \mathfrak{f}} = \frac{\log \varphi + \log x}{\log \mathfrak{f}} = \frac{\log x}{\log \mathfrak{f}} + \frac{\log \varphi}{\log \mathfrak{f}}
\]
\end{proof}

\subsection{Time Additivity}

\begin{theorem}[Time Additivity]
\label{thm:time-additivity}
\lean{FUST.FrourioLogarithm.time_additivity}
\leanok
For $x > 0$ and $n \in \mathbb{N}$:
\[
\log_{\mathfrak{f}}(\varphi^n \cdot x) = \log_{\mathfrak{f}}(x) + n \cdot \mathrm{phiStep}
\]
\end{theorem}

This shows that $n$ iterations of $\varphi$-scaling correspond to $n$ time steps in Frourio coordinates.
