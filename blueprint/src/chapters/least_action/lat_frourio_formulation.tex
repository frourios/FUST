\section{Lagrangian in Frourio Coordinates}
\label{sec:frourio-lagrangian}

\begin{definition}[FUST Lagrangian]
\label{def:fust-lagrangian}
\lean{FUST.LeastAction.FUSTLagrangian}
\leanok
In real space:
\[
L(f, x) := (D_6[f](x))^2
\]
\end{definition}

\begin{definition}[Frourio Lagrangian]
\label{def:frourio-lagrangian}
\lean{FUST.LeastAction.FrourioLagrangian}
\leanok
In Frourio coordinates:
\[
L_{\mathfrak{f}}(f, t) := (D_6[f](\mathfrak{f}^t))^2
\]
\end{definition}

\begin{theorem}[Lagrangian Non-negativity]
\label{thm:lagrangian-nonneg}
\lean{FUST.LeastAction.lagrangian_nonneg, FUST.LeastAction.frourio_lagrangian_nonneg}
\leanok
\[
L(f, x) \ge 0 \quad \text{and} \quad L_{\mathfrak{f}}(f, t) \ge 0
\]
\end{theorem}

\begin{theorem}[Coordinate Change]
\label{thm:lagrangian-coord-change}
\lean{FUST.LeastAction.lagrangian_coordinate_change}
\leanok
For $x > 0$:
\[
L(f, x) = L_{\mathfrak{f}}(f, \log_{\mathfrak{f}}(x))
\]
\end{theorem}

\section{The Beautiful Reformulation}
\label{sec:beautiful-reformulation}

In Frourio coordinates, the least action theorem becomes:

\begin{enumerate}
    \item \textbf{Time = Frourio coordinate}: $t = \log_{\mathfrak{f}}(x)$
    \item \textbf{Lagrangian}: $L(t) = (D_6[f](\mathfrak{f}^t))^2$ is uniform in $t$
    \item \textbf{Action}: $A = \int L(t) \, dt$ measures deviation from timelessness
    \item \textbf{Timeless states}: $L(t) = 0$ for all $t$ $\Leftrightarrow$ $f \in \ker(D_6)$
    \item \textbf{Massive states}: $\exists t: L(t) > 0$ $\Leftrightarrow$ proper time exists
\end{enumerate}

\begin{theorem}[Complete Frourio Formulation]
\label{thm:frourio-least-action}
\lean{FUST.LeastAction.frourio_least_action_formulation}
\leanok
\begin{enumerate}
    \item[(A)] Coordinate change preserves Lagrangian
    \item[(B)] $\ker(D_6)$ states have zero action density
    \item[(C)] Positive action density implies time exists
    \item[(D)] $\varphi$-scaling becomes time translation
\end{enumerate}
\end{theorem}

\begin{theorem}[Time is Logarithmic Deviation]
\label{thm:time-log-deviation}
\lean{FUST.LeastAction.time_is_log_deviation}
\leanok
\begin{enumerate}
    \item[(A)] Frourio time is logarithmic
    \item[(B)] Time step is uniform in Frourio space
    \item[(C)] Timeless states have zero Lagrangian
\end{enumerate}
\end{theorem}

\label{thm:frourio-time-equiv}
\lean{FUST.LeastAction.frourio_time_equivalence}

\end{enumerate}
\end{theorem}

\section{Physical Interpretation}
\label{sec:physical-interp-frourio}

\subsection{Action as Path Integral}

The action integral $\int L \, dx/x$ becomes $\int L \, dt$ in Frourio coordinates.

Key insight: In Frourio space, $\varphi$-scaling symmetry becomes translation symmetry, and the action integral is translation-invariant.

\subsection{The Essence}

\begin{quote}
\textbf{Time is the logarithmic deviation from scale invariance.}
\end{quote}

\begin{itemize}
    \item Frourio time is logarithmic
    \item Time step is uniform in Frourio space
    \item Timeless states ($\ker(D_6)$) have zero Lagrangian everywhere
    \item Massive states have positive Lagrangian somewhere
\end{itemize}
