\section{Main Theorem}

\begin{theorem}[Frourio Universal Structure Theorem]
\label{thm:fust}
The ring $\GEI$ with its natural Galois group
$\mathrm{Gal}(\Q(\sqrt5,\sqrt{-3})/\Q) \cong (\Z/2)^2$
uniquely determines:
\begin{enumerate}
  \item a 4-dimensional Lorentzian spacetime with signature $(1,3)$
    and symmetry group $\iso(3,1) = \so(3,1) \ltimes \R^4$,
  \item gauge groups $SU(3) \times SU(2) \times U(1)$
    from the Galois decomposition of $\GEI$,
  \item a positive Casimir mass gap $m^2 = 14 > 0$
    from the Poincar\'{e} structure of $\Dz$.
\end{enumerate}
No free parameters are involved:
every physical constant is a number-theoretic invariant of $\GEI$.
\end{theorem}

\section{Proof Outline}

The proof proceeds in six steps.
\begin{enumerate}
  \item \textbf{Algebraic foundations} (Chapter~\ref{ch:algebra}).
    Define $\vp$, $\ps$, $\zs$ and construct $\GEI$.
    The Galois automorphisms $\sigma$ and $\tau$ act on $\GEI$.
  \item \textbf{$\Fz$ operator} (Chapter~\ref{ch:fz}).
    The difference operators $\mathrm{Diff}_k$ have coefficients
    uniquely determined by kernel conditions.
    Construct $\Dz$ and its integral form $\Fz = 5z\,\Dz$.
    The mod~6 selection rule separates kernel from active modes.
  \item \textbf{Spacetime uniqueness} (Chapter~\ref{ch:spacetime}).
    The AF-channel ($\re(\afcoeff) = 0$, $\im(\afcoeff) > 0$)
    gives 1 temporal dimension.
    The SY-channel ($\phiS$ rank 3) gives 3 spatial dimensions.
    In log-coordinates, $\vp$-dilation becomes translation:
    $\exp(t + \log\vp) = \vp\exp(t)$.
    The resulting symmetry group is the Poincar\'{e} group $\iso(3,1)$
    with $\dim = 10$.
  \item \textbf{Gauge groups} (Chapter~\ref{ch:gauge}).
    $U(1)$: Galois-fixed subring of $\GEI$ is $\Z$ (rank 1).
    $SU(2)$: $\tau$-anti-invariance and $\afcoeff^2_{\mathrm{gei}} = -12$
    (quaternionic).
    $SU(3)$: $\Z[\zs]$ has 6 units forming the $A_2$ root system ($\det = 3$).
  \item \textbf{Mass gap} (Chapter~\ref{ch:mass}).
    Casimir mass squared $m^2 = 14$.
  \item \textbf{Galois coarsening} (Chapter~\ref{ch:coarsening}).
    The three nontrivial Galois elements define canonical projections
    $N_\tau$, $N_\sigma$, $N$ forming a commutative diagram.
    $\tau$-norm formula: $|\lambda|^2 = \beta^2 + 12\alpha^2$.
    Decomposition: $|\lambda|^2 = 300\,m^2 + 300\sum\sigma_i^2 + 36\,\phiS^2$.
\end{enumerate}

\section{Applications}

The structure theorem has the following consequences.
\begin{itemize}
  \item \textbf{Thermodynamics} (Chapter~\ref{ch:thermo}).
    The entropy $S(f)(z) := |\Fz f(z)|^2$ and time evolution
    $f \mapsto f(\vp\,\cdot\,)$ yield the three laws of thermodynamics.
  \item \textbf{Navier--Stokes global regularity}
    (Chapter~\ref{ch:ns}).
    Dissipation coefficients $C_n$ from the AF-channel spectral structure,
    Planck-scale truncation, and energy decay.
  \item \textbf{Yang--Mills mass gap}
    (Chapter~\ref{ch:ym}).
    The derived gauge group structure combined with $m^2 > 0$.
\end{itemize}
