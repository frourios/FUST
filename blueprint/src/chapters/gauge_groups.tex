\begin{theorem}[$\varphi$-Dilation Commutant is Diagonal]
\label{thm:commutant_diagonal}
\lean{FUST.phiDilation_commutant_diagonal}
\leanok
Let $S = \mathrm{diag}(1, \varphi, \varphi^2)$ be the scaling matrix on the syWeight $= 3$ space. If $M \in \mathrm{GL}(3, \mathbb{C})$ satisfies $MS = SM$, then $M$ is diagonal.

View $V_0 := \mathrm{span}\{1, z\}$ (AF channel, dim $= \mathrm{spinDegeneracy} = 2$) with basis $|{\uparrow}\rangle = 1$, $|{\downarrow}\rangle = z$.
