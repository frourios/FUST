\section{Generation Constraint}
\label{sec:particle_generations}

\begin{theorem}[Maximum Generations]
\label{thm:particle_max_generations}
\lean{FUST.ParticleSpectrum.maxGenerations_from_kernel_transition}
\leanok
Maximum generations = 3, derived from kernel dimension transition:
\[
\text{maxGenerations} = (C(3,2) - C(2,2)) + 1 = (3 - 1) + 1 = 3
\]
\end{theorem}

\begin{theorem}[$D_3$, $D_4$ Selection]
\label{thm:particle_d3_d4_selection}
\lean{FUST.ParticleSpectrum.selected_D_structures_kernel_dim_1}
\leanok
$D_3$ and $D_4$ are selected because they have kernel dimension 1:
\begin{itemize}
\item $D_3$ does NOT annihilate linear: $\exists x \neq 0: D_3[x](x) \neq 0$
\item $D_4$ does NOT annihilate linear: $\exists x \neq 0: D_4[x](x) \neq 0$
\end{itemize}
\end{theorem}

\begin{theorem}[Fourth Generation Forbidden]
\label{thm:particle_4th_gen_forbidden}
\lean{FUST.ParticleSpectrum.fourth_generation_forbidden}
\leanok
4th generation would require $D_7$, which projects to $D_6$.
\end{theorem}

\section{SM Particle Count}
\label{sec:particle_count}

\begin{theorem}[Fermion Count]
\label{thm:particle_fermion_count}
\lean{FUST.ParticleSpectrum.SM_fermion_count_eq}
\leanok
\begin{align}
\text{Leptons} &= 2 \times 3 = 6 \\
\text{Quarks} &= 2 \times 3 \times C(3,2) = 18 \\
\text{Total fermions} &= 24
\end{align}
\end{theorem}

\begin{theorem}[Boson Count]
\label{thm:particle_boson_count}
\lean{FUST.ParticleSpectrum.SM_boson_count_eq}
\leanok
\begin{align}
\text{Gluons} &= C(3,2)^2 - 1 = 8 \\
\text{Weak bosons (W}^\pm\text{, Z)} &= C(3,2) = 3 \\
\text{Photon} &= 1 \\
\text{Higgs} &= 1 \\
\text{Total bosons} &= 13
\end{align}
\end{theorem}

\begin{theorem}[Total SM Particles]
\label{thm:particle_sm_count}
\lean{FUST.ParticleSpectrum.SM_count}
\leanok
\[
\text{SM particles} = 24 + 13 = 37
\]
Including graviton: 38.
\end{theorem}
