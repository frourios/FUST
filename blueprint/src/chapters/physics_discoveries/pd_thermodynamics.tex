\section{Zeroth Law: Thermal Equilibrium}
\label{sec:thermo_zeroth_law}

In standard thermodynamics, the zeroth law establishes that thermal equilibrium is
an equivalence relation. In FUST, $\ker(F_\zeta)$ states are all at ``absolute zero''
--- they share identical entropy $S \equiv 0$.

\begin{theorem}[Zeroth Law: Kernel Equilibrium]
\label{thm:zeroth_law}
\lean{FUST.Thermodynamics.zeroth_law_kernel_equilibrium}
\leanok
For $f, g \in \ker(F_\zeta)$:
\[
\forall z,\, S(f)(z) = S(g)(z) = 0
\]
\end{theorem}

For states outside $\ker(F_\zeta)$, same-degree scalar multiples have
gauge-invariant entropy ratio, enabling temperature comparison:

\begin{theorem}[Zeroth Law: Same-Degree Thermal Ratio]
\label{thm:zeroth_law_thermal}
\lean{FUST.Thermodynamics.zeroth_law_same_degree_thermal}
\leanok
For $F_\zeta f(z) \neq 0$:
\[
\frac{S(a \cdot f)(z)}{S(b \cdot f)(z)} = \frac{|a|^2}{|b|^2}
\]
independent of $z$ (gauge-invariant).
\end{theorem}

\section{First Law: Energy Conservation}
\label{sec:thermo_first_law}

Energy conservation follows from $\varphi$-equivariance of the $F_\zeta$ Lagrangian.
This is the FUST Noether current: $\varphi$-scale symmetry implies energy conservation.

\begin{theorem}[First Law: $\varphi$-Equivariance]
\label{thm:first_law_equivariance}
\lean{FUST.Thermodynamics.first_law_equivariance}
\leanok
\[
L[\mathrm{timeEvolution}(f)](z) = L[f](\varphi z)
\]
\end{theorem}

\begin{theorem}[Vacuum Energy is Zero]
\label{thm:first_law_vacuum}
\lean{FUST.Thermodynamics.first_law_vacuum_energy}
\leanok
For $f \in \ker(F_\zeta)$, $\forall n \in \mathbb{Z}$:
\[
S[f](n) = |F_\zeta f(\varphi^n)|^2 = 0
\]
\end{theorem}

\begin{theorem}[Energy Non-negativity]
\label{thm:first_law_nonneg}
\lean{FUST.Thermodynamics.first_law_energy_nonneg}
\leanok
\[
\mathcal{A}_N[f] \geq 0
\]
\end{theorem}

\begin{theorem}[Action Monotonicity]
\label{thm:first_law_monotone}
\lean{FUST.Thermodynamics.first_law_action_monotone}
\leanok
For $N \leq M$:
\[
\sum_{n=-N}^{N} |F_\zeta f(\varphi^n)|^2 \leq \sum_{n=-M}^{M} |F_\zeta f(\varphi^n)|^2
\]
Enlarging the scale window cannot decrease energy.
\end{theorem}
